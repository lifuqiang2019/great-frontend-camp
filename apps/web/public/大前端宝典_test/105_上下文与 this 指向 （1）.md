# 上下文与 this 指向 （1）

```javascript
var name = 'window'  
  
const person1 = {  
name: 'person1',  
foo1: function() {  
console.log(this.name)  
},  
foo2: () => {  
console.log(this.name)  
},  
foo3: function() {  
return function() {  
console.log(this.name)  
}  
}  
}  
  
const person2 = {  
name: 'person2'  
}  
person1.foo1() // person1  
person1.foo1.call(person2) // person2  
  
person1.foo2() // window  
person1.foo2.call(person2) // window  
  
person1.foo3()() // window  
person1.foo3.call(person2)() // window

```
