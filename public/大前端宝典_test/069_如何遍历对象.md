# 如何遍历对象

```javascript
// for in  
const obj = { a: 1, b: 2, c: 3 };  
for (let key in obj) {  
console.log(key, obj\[key\]);  
}  
  
// Object.keys  
const obj = { a: 1, b: 2, c: 3 };  
const keys = Object.keys(obj);  
keys.forEach(key => {  
console.log(key, obj\[key\]);  
});  
  
// Object.entries  
const obj = { a: 1, b: 2, c: 3 };  
const entries = Object.entries(obj);  
entries.forEach((\[key, value\]) => {  
console.log(key, value);  
});
```
