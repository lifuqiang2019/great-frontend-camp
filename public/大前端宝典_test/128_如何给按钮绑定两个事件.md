# 如何给按钮绑定两个事件

```javascript
const button = document.querySelector('button');  
  
function handleClick1() {  
console.log('点击事件1');  
}  
  
function handleClick2() {  
console.log('点击事件2');  
}  
  
button.addEventListener('click', handleClick1);  
button.addEventListener('click', handleClick2);

```
