# 实现一个柯里化函数 add

1.  add(1,2,3).valueOf() // 6
2.  add(1,2)(3)(4,5).valueOf() // 15
3.  add(1)(2)(3)(4,5,6)(7).valueOf() // 28

```javascript
const curry = (...args1) => {  
let params = args1;  
  
const addFn = (...args2) => {  
params = params.concat(args2);  
return addFn;  
}  
  
addFn.valueOf = () => {  
return params.reduce((pre, cur) => {  
return pre + cur;  
}, 0);  
}  
  
return addFn;  
}

```
