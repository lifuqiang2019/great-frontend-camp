# 找出字符串中不含有重复字符的 最长子串 的长度。

```javascript
var lengthOfLongestSubstring = function(s) {  
let arr = \[\];  
let max = 0;  
for (let i = 0, len = s.length; i < len; ++i) {  
const sameIndex = arr.findIndex(item => item === s\[i\]);  
arr.push(s\[i\]);  
if (sameIndex > -1) {  
arr = arr.splice(sameIndex + 1);  
}  
max = Math.max(arr.length, max);  
}  
return max;  
};

```
